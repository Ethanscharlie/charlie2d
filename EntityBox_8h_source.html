244 slideOut slide(const Vector2f &move, const std::vector<Entity *> &solids,
245 bool pushOut = true, bool useMoveBox = true) {
246 slideOut out;
247
248 // Start
249 Vector2f movement = move;
250
251 // Create Horizonal check box
252 // The check box is sized to the move to prevent clipping
253 // at high speeds and low framerates
254 changePosition({movement.x, 0});
255 if (!useMoveBox)
256 getBox() = getBox();
257
258 // Horizontal check and move
259 HORREDO:
260 for (Entity *col : solids) {
261 if (!getBox().checkCollision(col->box->getBox()))
262 continue;
263
264 out.hitList.push_back(col);
265 if (!pushOut)
266 continue;
267
268 Entity *other = col;
269 if (movement.x > 0) {
270 setXPosition(other->box->getBox().getLeft() - getSize().x);
271 out.horizontalHit = true;
272 } else if (movement.x < 0) {
273 setXPosition(other->box->getBox().getRight());
274 out.horizontalHit = true;
275 }
276
277 goto HORREDO;
278 }
279
280 // Create verticle check box
281 changePosition({0, movement.y});
282 if (!useMoveBox)
283 getBox() = getBox();
284
285 // Verticle collision check and move
286 VERREDO:
287 for (Entity *col : solids) {
288 if (!getBox().checkCollision(col->box->getBox()))
289 continue;
290
291 out.hitList.push_back(col);
292 if (!pushOut)
293 continue;
294
295 Entity *other = col;
296 if (movement.y > 0) {
297 setYPosition(other->box->getBox().getTop() - getSize().y);
298 out.verticleHit = true;
299 } else if (movement.y < 0) {
300 setYPosition(other->box->getBox().getBottom());
301 out.verticleHit = true;
302 }
303
304 goto VERREDO;
305 }
306
307 return out;
308 }
