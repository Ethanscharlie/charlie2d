239 slideOut slide(const Vector2f &move, const std::vector<Entity *> &solids,
240 bool pushOut = true, bool useMoveBox = true) {
241 slideOut out;
242
243 // Start
244 Vector2f movement = move;
245
246 // This system checks collisions for horizontal and vericle movment
247 // seperatly If there is a collsion the entityBox can be moved in the
248 // opposite direction outside the collided box
249 // Collsions are restarted if there is any colllsion at all to prevent
250 // clipping
251
252 // Create Horizonal check box
253 // The check box is sized to the move to prevent clipping
254 // at high speeds and low framerates
255 changePosition({movement.x, 0});
256 if (!useMoveBox)
257 getBox() = getBox();
258
259 // Horizontal check and move
260 HORREDO:
261 for (Entity *col : solids) {
262 if (!getBox().checkCollision(col->require<entityBox>()->getBox()))
263 continue;
264
265 out.hitList.push_back(col);
266 if (!pushOut)
267 continue;
268
269 Entity *other = col;
270 if (movement.x > 0) {
271 setXPosition(other->require<entityBox>()->getBox().getLeft() - getSize().x);
272 out.horizontalHit = true;
273 } else if (movement.x < 0) {
274 setXPosition(other->require<entityBox>()->getBox().getRight());
275 out.horizontalHit = true;
276 }
277
278 goto HORREDO;
279 }
280
281 // Create verticle check box
282 changePosition({0, movement.y});
283 if (!useMoveBox)
284 getBox() = getBox();
285
286 // Verticle collision check and move
287 VERREDO:
288 for (Entity *col : solids) {
289 if (!getBox().checkCollision(col->require<entityBox>()->getBox()))
290 continue;
291
292 out.hitList.push_back(col);
293 if (!pushOut)
294 continue;
295
296 Entity *other = col;
297 if (movement.y > 0) {
298 setYPosition(other->require<entityBox>()->getBox().getTop() - getSize().y);
299 out.verticleHit = true;
300 } else if (movement.y < 0) {
301 setYPosition(other->require<entityBox>()->getBox().getBottom());
302 out.verticleHit = true;
303 }
304
305 goto VERREDO;
306 }
307
308 return out;
309 }
