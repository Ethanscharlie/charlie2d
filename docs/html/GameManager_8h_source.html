174std::vector<C *> GameManager::getComponents(std::string tag) {
175 std::vector<C *> hits;
176 auto it = components.find(typeid(C));
177
178 if (it != components.end()) {
179 for (auto component : it->second) {
180 // Static cast each Component* to C*
181 if (component->entityTag == tag || tag == "") {
182 hits.push_back(static_cast<C *>(component));
183 }
184 }
185 } else
186 components[typeid(C)];
187
